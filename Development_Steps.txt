Bước 6: Thay nội dung file Program.cs
	using System;
	using System.Windows.Forms;
	using Microsoft.Extensions.DependencyInjection;
	using LFC_Management.Application; // Nạp tầng xử lý nghiệp vụ
	namespace LFC_Management.WinForm
	{
  	  internal static class Program
   	 {
       	 // Biến toàn cục để quản lý việc cấp phát các Handler sau này
        	public static IServiceProvider ServiceProvider { get; private set; }

        	[STAThread]
       	 	static void Main()
        	{
            		ApplicationConfiguration.Initialize();
            		// 1. Khởi tạo thùng chứa dịch vụ (DI Container)
           		 var services = new ServiceCollection();
            		// 2. Kích hoạt MediatR quét các Command/Query ở tầng Application
            		services.AddApplicationServices();
            		// 3. Đăng ký Form1 vào hệ thống DI Container
           		 services.AddTransient<Form1>();
          		  // Xây dựng bộ nạp dịch vụ
            		ServiceProvider = services.BuildServiceProvider();
           		 // 4. Gọi Form1 thông qua ServiceProvider thay vì dùng 'new Form1()'
            		// Viết tường minh 'System.Windows.Forms.Application' để tránh bị trùng tên với tầng Application của bạn
            		var mainForm = ServiceProvider.GetRequiredService<Form1>();
            		System.Windows.Forms.Application.Run(mainForm);
       		 }
   	 }
	}

Bước 8.1: Khai báo Connection String và kết nối các tầng trong Program.cs
Bước 8.2:Bạn vào project LFC_Management.Infrastructure, tạo một class file mới đặt tên là DesignTimeDbContextFactory.cs nằm ngay ở thư mục gốc của project này (chung chỗ với file DependencyInjection.cs), 
*** Lý do phải tạo file này: Khi bạn gõ lệnh Add-Migration, công cụ EF Core Tools sẽ tự động quét qua tầng Khởi chạy (WinForm) để tìm cấu hình DbContextOptions. Tuy nhiên, vì chúng ta đang cấu hình nạp chuỗi kết nối thông qua một hàm mở rộng (services.AddInfrastructureServices(connectionString)) và xây dựng ServiceProvider thủ công trong hàm Main của WinForms, công cụ Migration chạy ở môi trường Design-time (lúc chưa bấm Start App) không thể tự chạy vào hàm Main để nhặt cái cấu hình đó được. Hệ thống bị "mất phương hướng" và báo lỗi không thể khởi tạo LFCDbContext.
	Để giải quyết triệt để vấn đề này, EF Core cung cấp một giải pháp rất sạch sẽ: Chúng ta sẽ tạo một file mang tên "Nhà máy sản xuất DbContext lúc thiết kế" (IDesignTimeDbContextFactory). Khi bạn gõ lệnh Migration, EF Core Tools sẽ tự động tìm đến file này để lấy chuỗi kết nối và tạo DB mà không cần làm phiền đến tầng WinForms nữa.
Mở file Program.cs của project LFC_Management.WinForm ra. Chúng ta sẽ thêm chuỗi kết nối vào SQL Server cục bộ (LocalDB hoặc SQL Express) của bạn, sau đó truyền nó vào hàm AddInfrastructureServices mà chúng ta đã chuẩn bị ở tầng Hạ tầng.
